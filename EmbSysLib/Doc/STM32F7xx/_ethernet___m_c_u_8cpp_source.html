503 FrameTypeDef cHwEthernet_MCU::ETH_Get_Received_Frame(void)
504 {
505  uint32_t framelength = 0;
506  FrameTypeDef frame = {0,0,0};
507 
508  /* Get the Frame Length of the received packet: substruct 4 bytes of the CRC */
509  framelength = ((DMARxDescToGet->Status & ETH_DMARxDesc_FL) >> ETH_DMARxDesc_FrameLengthShift) - 4;
510  frame.length = framelength;
511 
512  /* Get the address of the buffer start address */
513  /* Check if more than one segment in the frame */
514  if (DMA_RX_FRAME_infos->Seg_Count >1)
515  {
516  frame.buffer =(DMA_RX_FRAME_infos->FS_Rx_Desc)->Buffer1Addr;
517  }
518  else
519  {
520  frame.buffer = DMARxDescToGet->Buffer1Addr;
521  }
522 
523  frame.descriptor = DMARxDescToGet;
524 
525  /* Update the ETHERNET DMA global Rx descriptor with next Rx descriptor */
526  /* Chained Mode */
527  /* Selects the next DMA Rx descriptor list for next buffer to read */
528  DMARxDescToGet = (ETH_DMADESCTypeDef*) (DMARxDescToGet->Buffer2NextDescAddr);
529 
530  /* Return Frame */
531  return (frame);

535 uint32_t cHwEthernet_MCU::ETH_Prepare_Transmit_Descriptors(uint16_t FrameLength)
536 {
537  uint32_t buf_count =0, size=0,i=0;
538 
539  /* Check if the descriptor is owned by the ETHERNET DMA (when set) or CPU (when reset) */
540  if((DMATxDescToSet->Status & ETH_DMATxDesc_OWN) != (uint32_t)RESET)
541  {
542  /* Return ERROR: OWN bit set */
543  return ETH_ERROR;
544  }
545 
546  if (FrameLength > ETH_TX_BUF_SIZE)
547  {
548  buf_count = FrameLength/ETH_TX_BUF_SIZE;
549  if (FrameLength%ETH_TX_BUF_SIZE) buf_count++;
550  }
551  else buf_count =1;
552 
553  if (buf_count ==1)
554  {
555  /*set LAST and FIRST segment */
556  DMATxDescToSet->Status |= (ETH_DMATxDesc_FS|ETH_DMATxDesc_LS);
557  /* Set frame size */
558  DMATxDescToSet->ControlBufferSize = (FrameLength& ETH_DMATxDesc_TBS1);
559  /* Set Own bit of the Tx descriptor Status: gives the buffer back to ETHERNET DMA */
560  DMATxDescToSet->Status |= ETH_DMATxDesc_OWN;
561  DMATxDescToSet= (ETH_DMADESCTypeDef *)(DMATxDescToSet->Buffer2NextDescAddr);
562  }
563  else
564  {
565  for (i=0; i< buf_count; i++)
566  {
567  /* Clear FIRST and LAST segment bits */
568  DMATxDescToSet->Status &= ~(ETH_DMATxDesc_FS | ETH_DMATxDesc_LS);
569  if (i==0)
570  {
571  /* Setting the first segment bit */
572  DMATxDescToSet->Status |= ETH_DMATxDesc_FS;
573  }
574 
575  /* Program size */
576  DMATxDescToSet->ControlBufferSize = (ETH_TX_BUF_SIZE & ETH_DMATxDesc_TBS1);
577 
578  if (i== (buf_count-1))
579  {
580  /* Setting the last segment bit */
581  DMATxDescToSet->Status |= ETH_DMATxDesc_LS;
582  size = FrameLength - (buf_count-1)*ETH_TX_BUF_SIZE;
583  DMATxDescToSet->ControlBufferSize = (size & ETH_DMATxDesc_TBS1);
584  }
585 
586  /*give back descriptor to DMA */
587  DMATxDescToSet->Status |= ETH_DMATxDesc_OWN;
588 
589  DMATxDescToSet = (ETH_DMADESCTypeDef *)(DMATxDescToSet->Buffer2NextDescAddr);
590  /* Set Own bit of the Tx descriptor Status: gives the buffer back to ETHERNET DMA */
591  }
592  }
593 
594  /* When Tx Buffer unavailable flag is set: clear it and resume transmission */
595  if ((ETH->DMASR & ETH_DMASR_TBUS) != (uint32_t)RESET)
596  {
597  /* Clear TBUS ETHERNET DMA flag */
598  ETH->DMASR = ETH_DMASR_TBUS;
599  /* Resume DMA transmission*/
600  ETH->DMATPDR = 0;
601  }
602 
603  /* Return SUCCESS */
604  return ETH_SUCCESS;
605 }
