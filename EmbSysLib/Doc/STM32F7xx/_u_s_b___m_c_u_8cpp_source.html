463 
464 // USB_OTG_FS->GCCFG |= USB_OTG_GCCFG_VBUSBSEN; // Activate VBUS Sensing B
465 // USB_OTG_FS->GCCFG |= USB_OTG_GCCFG_NOVBUSSENS; // No Vbus sensing
466  USB_PCGCCTL = 0; // Restart the Phy Clock

517 //-------------------------------------------------------------------
518 inline void cHwUSB_0::isr(void)
519 {
520  // Current mode of operation = Host mode ?
521  if( USB_OTG_FS->GINTSTS & USB_OTG_GINTSTS_CMOD )
522  {
523  return;
524  }
525 
526  //-----------------------------------------------------------------
527  if( isInterruptPending( USB_OTG_GINTSTS_MMIS ) )
528  {
529  // incorrect mode
530  clrInterrupt( USB_OTG_GINTSTS_MMIS);
531  }
532 
533  //-----------------------------------------------------------------
534  if( isInterruptPending( USB_OTG_GINTSTS_OEPINT ) )
535  {
536  DWORD interruptBits = getOutInterrupt();
537  DWORD interruptType = 0;
538  BYTE epNum = 0;
539 
540  while ( interruptBits )
541  {
542  if( interruptBits & 0x1 )
543  {
544  interruptType = epOUT[epNum].getInterrupt();
545 
546  if( interruptType & USB_OTG_DOEPINT_XFRC )
547  {
548  // OUT-package received, non-control
549  epOUT[epNum].receive();
550  eventHandler( epNum, false );
551  epOUT[epNum].clrInterrupt( USB_OTG_DOEPINT_XFRC );
552  }
553 
554  if( interruptType & USB_OTG_DOEPINT_STUP )
555  {
556  // Setup-package received
557  eventHandler( 0, true );
558  epIN[0].transmit(true);
559  epOUT[epNum].clrInterrupt( USB_OTG_DOEPINT_STUP );
560  }
561 
562  if( interruptType & USB_OTG_DOEPINT_OTEPDIS )
563  {
564  epOUT[epNum].clrInterrupt( USB_OTG_DOEPINT_OTEPDIS);
565  }
566  }
567 
568  // handle next endpoint
569  epNum++;
570  interruptBits >>= 1;
571  }
572  }
573 
574  //-----------------------------------------------------------------
575  if( isInterruptPending( USB_OTG_GINTSTS_IEPINT ) )
576  {
577  DWORD interruptBits = getInInterrupt();
578  DWORD interruptType = 0;
579  BYTE epNum = 0;
580 
581  while( interruptBits )
582  {
583  if( interruptBits & 0x1 )
584  {
585  interruptType = epIN[epNum].getInterrupt( );
586 
587  if( interruptType & USB_OTG_DIEPINT_XFRC )
588  {
589  USB_DEVICE->DIEPEMPMSK = ~(0x1 << epNum);
590  epIN[epNum].clrInterrupt( USB_OTG_DIEPINT_XFRC );
591  epOUT[0].receiveZLP();
592  eventHandler( epNum | 0x80, true );
593  epIN[epNum].transmit( false );
594  }
595 
596  if( interruptType & USB_OTG_DIEPINT_TOC )
597  {
598  epIN[epNum].clrInterrupt( USB_OTG_DIEPINT_TOC );
599  }
600 
601  if( interruptType & USB_OTG_DIEPINT_ITTXFE )
602  {
603  epIN[epNum].clrInterrupt( USB_OTG_DIEPINT_ITTXFE );
604  }
605 
606  if( interruptType & USB_OTG_DIEPINT_INEPNE )
607  {
608  epIN[epNum].clrInterrupt( USB_OTG_DIEPINT_INEPNE );
609  }
610 
611  if( interruptType & USB_OTG_DIEPINT_EPDISD )
612  {
613  epIN[epNum].clrInterrupt( USB_OTG_DIEPINT_EPDISD );
614  }
615 
616  if( interruptType & USB_OTG_DIEPINT_TXFE )
617  {
618  epIN[epNum].writeToFifo();
619  epIN[epNum].clrInterrupt( USB_OTG_DIEPINT_TXFE );
620  }
621  }
622 
623  // handle next endpoint
624  epNum++;
625  interruptBits >>= 1;
626  }
627  }
628 
629  //-----------------------------------------------------------------
630  if( isInterruptPending( USB_OTG_GINTSTS_WKUINT ) )
631  {
632  USB_DEVICE->DCTL &= ~USB_OTG_DCTL_RWUSIG;
633  clrInterrupt( USB_OTG_GINTSTS_WKUINT );
634  }
635 
636  //-----------------------------------------------------------------
637  if( isInterruptPending( USB_OTG_GINTSTS_USBSUSP ) )
638  {
639  clrInterrupt( USB_OTG_GINTSTS_USBSUSP );
640  }
641 
642  //-----------------------------------------------------------------
643  if( isInterruptPending( USB_OTG_GINTSTS_USBRST ) )
644  {
645  reset();
646  clrInterrupt( USB_OTG_GINTSTS_USBRST );
647  }
648 
649  //-----------------------------------------------------------------
650  if( isInterruptPending( USB_OTG_GINTSTS_ENUMDNE ) )
651  {
652  USB_DEVICE->DCTL |= USB_OTG_DCTL_CGINAK;
653 
654  USB_OTG_FS->GUSBCFG &= ~( USB_OTG_GUSBCFG_TRDT );
655  USB_OTG_FS->GUSBCFG |= ( USB_OTG_GUSBCFG_TRDT_0
656  | USB_OTG_GUSBCFG_TRDT_2 );
657 
658  epOUT[0].config( MAX_EP0_SIZE );
659  epIN [0].config( MAX_EP0_SIZE );

661  clrInterrupt( USB_OTG_GINTSTS_ENUMDNE );

664  //-----------------------------------------------------------------
