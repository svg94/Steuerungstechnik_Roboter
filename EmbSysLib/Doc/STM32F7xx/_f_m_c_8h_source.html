139  void SDRAM_Init( )
140  {
141  uint32_t reg = 0;
142 
143  // Step 1 and 2
144  // Common initialization,
145  // see Reference Manual, DocID028270-Rev 3, page 388
146  //--------------------------------------------------
147  // SDRAM Control registers
148  reg = FMC_Bank5_6->SDCR[0/*BANK1*/];
149  reg &= ~( FMC_SDCR1_SDCLK
150  | FMC_SDCR1_RBURST
151  | FMC_SDCR1_RPIPE );
152  reg |= ( (2<<10) // Clock configuration: 2xHCLK periods
153  | FMC_SDCR1_RBURST // Burst read:enable
154  | (0<<13) ); // Read pipe: No HCLK clock cycle delay
155  FMC_Bank5_6->SDCR[0/*BANK1*/] = reg;
156 
157  // SDRAM Timing registers
158  reg = FMC_Bank5_6->SDTR[0/*BANK1*/];
159  reg &= ~( FMC_SDTR1_TRC
160  | FMC_SDTR1_TRP );
161  reg |= ( ((7 - 1) << 12) // Row cycle delay: 7 cycles
162  | ((2 - 1) << 20) ); // Row precharge delay: 2 cycles
163  FMC_Bank5_6->SDTR[0/*BANK1*/] = reg;
164 
165  // Bank specific initialization
166  //-----------------------------
167  // SDRAM Control registers
168  reg = FMC_Bank5_6->SDCR[bank];
169  reg &= ~( FMC_SDCR1_NC
170  | FMC_SDCR1_NR
171  | FMC_SDCR1_MWID
172  | FMC_SDCR1_NB
173  | FMC_SDCR1_CAS
174  | FMC_SDCR1_WP );
175  reg |= ( (0 << 0) // Number of column address bits:8
176  | (1 << 2) // Number of row address bits:12
177  | (2 << 4) // Memory data bus width:32 bits
178  | (1 << 6) // Number of internal banks:4
179  | (3 << 7) // CAS Latency:3 (must be same as in Load Mode Register)
180  |!FMC_SDCR1_WP ); // Write protection: disable
181  FMC_Bank5_6->SDCR[bank] = reg;
182 
183  // SDRAM Timing registers
184  reg = FMC_Bank5_6->SDTR[bank];
185  reg &= ~( FMC_SDTR1_TMRD
186  | FMC_SDTR1_TXSR
187  | FMC_SDTR1_TRAS
188  | FMC_SDTR1_TWR
189  | FMC_SDTR1_TRCD );
190  reg |= ( ((2 - 1) << 0) // Load Mode Register to Active: 2 cycles
191  | ((7 - 1) << 4) // Exit Self-refresh delay: 7 cycles
192  | ((4 - 1) << 8) // Self refresh time: 4 cycles
193  | ((2 - 1) << 16) // Recovery delay: 2 cycles
194  | ((2 - 1) << 24)); // Row to column delay: 2 cycles
195  FMC_Bank5_6->SDTR[bank] = reg;

199  //-------------------------------

201  // Step 3: Configure a clock configuration enable command
202  SendCommand( 0x01/*Clock Configuration Enable*/, 1, 0 );
