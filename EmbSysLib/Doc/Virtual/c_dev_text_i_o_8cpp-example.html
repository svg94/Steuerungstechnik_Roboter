//*******************************************************************
//*******************************************************************
#include "lib.h"
#include "_config.h"
//*******************************************************************
int main( void )
{
com.printf( "\r\n\ncDevText_IO\r\n" );
com.printf( __DATE__ "\r\n\n" );
com.printf( __TIME__ "\r\n\n" );
while( 1 )
{
if( char *str = com.getString() )
{
switch( (BYTE)str[0] )
{
case 0xC1: com.printf( "<Ctrl up> \r\n" ); break;
case 0xC2: com.printf( "<Ctrl down> \r\n" ); break;
case 0xC3: com.printf( "<Ctrl right>\r\n" ); break;
case 0xC4: com.printf( "<Ctrl left> \r\n" ); break;
default: com.printf( "-->%s\r\n", str ); break;
}
}
}
}
