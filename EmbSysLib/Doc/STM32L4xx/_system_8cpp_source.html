176

179  | RCC_CR_CSSON
180  | RCC_CR_HSEON
181  | RCC_CR_HSION );
182 
183  // PLL configuration register
184  RCC->PLLCFGR = 0x00001000; // Reset value,
185  // see Reference manual
186 
187  // Clock control register
188  RCC->CR &= ~(RCC_CR_HSEBYP); // reset ...
189 
190  // Clock interrupt register
191  RCC->CIER = 0; // Disable all interrupts
192 
193  //-----------------------------------------------------------------
194  #if defined (_HSE_BYPASS_ON )
195  RCC->CR |= (RCC_CR_HSEON | RCC_CR_HSEBYP);
196  #elif defined (_HSE_BYPASS_OFF )
197  RCC->CR |= (RCC_CR_HSEON );
198  #else
199  #error "_HSE_BYPASS not defined"
200  #endif
201  //-----------------------------------------------------------------
203  // Clock control register
204  RCC->CR |= RCC_CR_MSION; // Need HSION in ADC (only?)
205 
206  // Wait until MSI is ready or timeout
207  for( unsigned i = 0; i < HSE_STARTUP_TIMEOUT; i++ )
208  {
209  if( RCC->CR & RCC_CR_MSIRDY )
210  break;
211  }
212 
213  if( RCC->CR & RCC_CR_MSIRDY )
214  {
215  // Flash Acess Control Register
216  FLASH->ACR = FLASH_ACR_ICEN // Instruction cache enable
217  | FLASH_ACR_DCEN // Data cache enable
218  | FLASH_ACR_LATENCY_4WS; // Latency: 4 wait states
219

222

224  PWR->CR1 |= (1<<14); // Regulator voltage scaling output selection:
225  // VOS=1: Scale 1 mode
226

229  RCC->CFGR |= RCC_CFGR_HPRE_DIV1 // AHB prescaler: HCLK = SYSCLK
230  | RCC_CFGR_PPRE2_DIV1 // APB high-speed prescaler (APB2): PCLK2 = HCLK/1
231  // PCLK2 must not exceed 84 MHz
232  | RCC_CFGR_PPRE1_DIV1; // APB low-speed prescaler (APB1): PCLK1 = HCLK/1
233  // PCLK1 must not exceed 42 MHz
234

240  +----+ +->| | | | +----+
241  | +-----+ | |
242  | | | +----+ f_pll48m1clk

248  +----+
