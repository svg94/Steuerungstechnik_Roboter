95  case WD_TIMEOUT_16ms: pr = 0; rlr = 160; break; // 40kHz/4*16ms
96  case WD_TIMEOUT_32ms: pr = 0; rlr = 320; break; // 40kHz/4*32ms
97  case WD_TIMEOUT_65ms: pr = 0; rlr = 650; break; // 40kHz/4*65ms
98  case WD_TIMEOUT_130ms: pr = 0; rlr = 1300; break; // 40kHz/4*130ms
99  case WD_TIMEOUT_260ms: pr = 0; rlr = 2600; break; // 40kHz/4*260ms
100  case WD_TIMEOUT_520ms: pr = 1; rlr = 2600; break; // 40kHz/8*520ms
101  case WD_TIMEOUT_1000ms: pr = 2; rlr = 2500; break; // 40kHz/16*1000ms
102  case WD_TIMEOUT_2000ms: pr = 3; rlr = 2500; break; // 40kHz/32*2000ms
103  }

105  IWDG->KR = 0x5555;

107  IWDG->KR = 0x5555;

166  // Clock control register
167  RCC->CR |= RCC_CR_MSION; // MSI clock enable: MSI oscillator ON
168 
169  // Clock configuration register
170  RCC->CFGR &= ~( RCC_CFGR_MCOPRE // reset ...
171  | RCC_CFGR_MCOSEL
172  | RCC_CFGR_PPRE1
173  | RCC_CFGR_PPRE2
174  | RCC_CFGR_HPRE
175  | RCC_CFGR_SW );
176 
177  // Clock control register
178  RCC->CR &= ~( RCC_CR_HSION // reset ...
179  | RCC_CR_HSEON
180  | RCC_CR_PLLON
181  | RCC_CR_CSSON );
182  RCC->CR &= ~( RCC_CR_HSEBYP ); // reset ...
183 
184  // Clock configuration register
185  RCC->CFGR &= ~( RCC_CFGR_PLLMUL // reset ...
186  | RCC_CFGR_PLLDIV
187  | RCC_CFGR_PLLSRC );
188 
189  // Clock interrupt register
190  RCC->CIR = 0; // Disable all interrupts
191 
192  //*****************************************************************
193  // Configure:
194  // - System clock frequency
195  // - AHB/APBx prescalers
196  // - Flash settings
197  //*****************************************************************
198 
199  //-----------------------------------------------------------------
200  #if defined (_HSE_BYPASS_ON )
201  RCC->CR |= (RCC_CR_HSEON | RCC_CR_HSEBYP);
202  #elif defined (_HSE_BYPASS_OFF )
203  RCC->CR |= (RCC_CR_HSEON );
204  #else
205  #error "_HSE_BYPASS not defined"
206  #endif
207 
208  // Clock control register
209  // Needed in ADC (only?):
210  RCC->CR |= RCC_CR_HSION;
211 
212  // Wait for HSE is ready or timeout
213  for( unsigned i = 0; i < HSE_STARTUP_TIMEOUT; i++ )
214  {
215  if( RCC->CR & RCC_CR_HSERDY )
216  break;
217  }
218 
219  if( RCC->CR & RCC_CR_HSERDY )
220  {
221  // Flash Acess Control Register
222  FLASH->ACR |= FLASH_ACR_ACC64; // 64-bit access: enable
223  FLASH->ACR |= FLASH_ACR_PRFTEN; // Prefetch: enable
224  FLASH->ACR |= FLASH_ACR_LATENCY; // Latency: one wait state
225 
226  // Advanced Peripheral Bus Enable Register
227  RCC->APB1ENR |= RCC_APB1ENR_PWREN; // Power interface clock: enable
228 
229  // Power Control Register
230  PWR->CR = PWR_CR_VOS_0; // Voltage scaling range selection: range 1 (1.8 V)
231  while( PWR->CSR & PWR_CSR_VOSF ); // Wait until voltage regulator is ready
232 
233  // Clock Configuration Register
234  RCC->CFGR |= RCC_CFGR_HPRE_DIV1; // AHB prescaler: HCLK = SYSCLK
235  RCC->CFGR |= RCC_CFGR_PPRE2_DIV1; // APB high-speed prescaler (APB2): PCLK2 = HCLK
236  RCC->CFGR |= RCC_CFGR_PPRE1_DIV1; // APB low-speed prescaler (APB1): PCLK1 = HCLK
237 
238  RCC->CFGR |= ( RCC_CFGR_PLLSRC_HSE // PLL entry clock source: HSE
239  | RCC_CFGR_PLLMUL12 // PLL multiplication factor: PLLVCO = PLL clock entry x 12
240  | RCC_CFGR_PLLDIV3); // PLL output division: PLL clock output = PLLVCO / 3
241 
242  // Clock control register
243  RCC->CR |= RCC_CR_PLLON; // PLL enable: ON
244  while( !(RCC->CR & RCC_CR_PLLRDY) );// Wait until PLL is ready
245 
246  // Clock Configuration Register
247  RCC->CFGR |= RCC_CFGR_SW_PLL; // System clock switch: PLL used as system clock
248 
249  // Wait for system clock switch is ready
250  while( (RCC->CFGR & RCC_CFGR_SWS) != RCC_CFGR_SWS_PLL );
251  }
252 
253  else
254  {
255  // Error: Wrong clock configuration
256  }
257 
258  // System Control Block
259  SCB->VTOR = FLASH_BASE | 0x00; // Vector Table Relocation in Internal FLASH
260 }
