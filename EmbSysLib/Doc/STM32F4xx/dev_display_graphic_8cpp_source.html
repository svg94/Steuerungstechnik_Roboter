197 void cDevDisplayGraphic::drawLine( WORD x0,
198  WORD y0,
199  WORD x1,
200  WORD y1,
201  WORD th,
202  WORD color )
203 {
204  long t;
205  int x;
206  int y;
207  int dx = (int)x1-x0;
208  int dy = (int)y1-y0;
209 
210  if( dy == 0 ) // horizontal
211  {
212  x0 = (x0 < x1) ? x0 : x1;
213  y0 = (y0 < y1) ? y0 : y1;
214  drawRectangle( x0, y0-th/2, abs(dx), th, color );
215  }
216  else if( dx == 0 ) // vertikal
217  {
218  x0 = (x0 < x1) ? x0 : x1;
219  y0 = (y0 < y1) ? y0 : y1;
220  drawRectangle(x0-th/2, y0, th, abs(dy), color );
221  }
222  else if( abs(dx) <= abs(dy) )
223  {
224  for( t = abs(dy); t >= 0; t-- ) // t = 1 downto 0
225  {
226  x = x0 + t*dx/abs(dy); // x = x0+dx downto x0
227  y = y0 + ( (dy>0) ? +t : -t ); // y = y0+dy downto y0
228  for( int i = th; i >= 0; i-- )
229  {
230  drawPixel( x+i, y, color );
231  }
232  }
233  }
234  else
235  {
236  for( t = abs(dx); t >= 0; t-- ) // t = 1 downto 0
237  {
238  x = x0 + ( (dx>0) ? +t : -t ); // x = x0+dx downto x0
239  y = y0 + t*dy/abs(dx); // y = y0+dy downto y0
240  for( int i = th; i >= 0; i-- )
