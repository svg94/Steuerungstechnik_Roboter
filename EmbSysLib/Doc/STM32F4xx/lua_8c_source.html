178 ** Message handler used to run all chunks
179 */
180 static int msghandler (lua_State *L) {
181  const char *msg = lua_tostring(L, 1);
182  if (msg == NULL) { /* is error object not a string? */
183  if (luaL_callmeta(L, 1, "__tostring") && /* does it have a metamethod */
184  lua_type(L, -1) == LUA_TSTRING) /* that produces a string? */
185  return 1; /* that is the message */
186  else
187  msg = lua_pushfstring(L, "(error object is a %s value)",
188  luaL_typename(L, 1));

191  return 1; /* return the traceback */

195 /*
196 ** Interface to 'lua_pcall', which sets appropriate message function
197 ** and C-signal handler. Used to run all chunks.
198 */
199 static int docall (lua_State *L, int narg, int nres) {
200  int status;
201  int base = lua_gettop(L) - narg; /* function index */
202  lua_pushcfunction(L, msghandler); /* push message handler */
203  lua_insert(L, base); /* put it under function and args */
204  globalL = L; /* to be available to 'laction' */
205  signal(SIGINT, laction); /* set C-signal handler */
206  status = lua_pcall(L, narg, nres, base);
207  signal(SIGINT, SIG_DFL); /* reset C-signal handler */
208  lua_remove(L, base); /* remove message handler from the stack */
209  return status;
210 }

219 /*
220 ** Create the 'arg' table, which stores all arguments from the
221 ** command line ('argv'). It should be aligned so that, at index 0,
222 ** it has 'argv[script]', which is the script name. The arguments
223 ** to the script (everything after 'script') go to positive indices;
224 ** other arguments (before the script name) go to negative indices.
225 ** If there is no script name, assume interpreter's name as base.
226 */
227 static void createargtable (lua_State *L, char **argv, int argc, int script) {
228  int i, narg;
229  if (script == argc) script = 0; /* no script name? */
230  narg = argc - (script + 1); /* number of positive indices */
231  lua_createtable(L, narg, script + 1);
232  for (i = 0; i < argc; i++) {
233  lua_pushstring(L, argv[i]);
234  lua_rawseti(L, -2, i - script);
235  }
236  lua_setglobal(L, "arg");
237 }

305 /*
306 ** Prompt the user, read a line, and push it into the Lua stack.
307 */
308 static int pushline (lua_State *L, int firstline) {
309  char buffer[LUA_MAXINPUT];
310  char *b = buffer;
311  size_t l;
312  const char *prmt = get_prompt(L, firstline);
313  int readstatus = lua_readline(L, b, prmt);
314  if (readstatus == 0)
315  return 0; /* no input (prompt will be popped by caller) */
316  lua_pop(L, 1); /* remove prompt */
317  l = strlen(b);
318  if (l > 0 && b[l-1] == '\n') /* line ends with newline? */
319  b[--l] = '\0'; /* remove it */
320  if (firstline && b[0] == '=') /* for compatibility with 5.2, ... */
321  lua_pushfstring(L, "return %s", b + 1); /* change '=' to 'return' */
322  else
323  lua_pushlstring(L, b, l);
324  lua_freeline(L, b);
325  return 1;
326 }

329 /*
330 ** Try to compile line on the stack as 'return <line>;'; on return, stack
331 ** has either compiled chunk or original line (if compilation failed).
332 */
333 static int addreturn (lua_State *L) {
334  const char *line = lua_tostring(L, -1); /* original line */
335  const char *retline = lua_pushfstring(L, "return %s;", line);
336  int status = luaL_loadbuffer(L, retline, strlen(retline), "=stdin");
337  if (status == LUA_OK) {
338  lua_remove(L, -2); /* remove modified line */
339  if (line[0] != '\0') /* non empty? */
340  lua_saveline(L, line); /* keep history */
341  }
342  else
343  lua_pop(L, 2); /* pop result from 'luaL_loadbuffer' and modified line */
344  return status;
345 }

348 /*
349 ** Read multiple lines until a complete Lua statement
350 */
351 static int multiline (lua_State *L) {
352  for (;;) { /* repeat until gets a complete statement */
353  size_t len;
354  const char *line = lua_tolstring(L, 1, &len); /* get what it has */
355  int status = luaL_loadbuffer(L, line, len, "=stdin"); /* try it */
356  if (!incomplete(L, status) || !pushline(L, 0)) {
357  lua_saveline(L, line); /* keep history */
358  return status; /* cannot or should not try to add continuation line */
359  }
360  lua_pushliteral(L, "\n"); /* add newline... */
361  lua_insert(L, -2); /* ...between the two lines */
362  lua_concat(L, 3); /* join them */
363  }
364 }

368 ** Read a line and try to load (compile) it first as an expression (by
369 ** adding "return " in front of it) and second as a statement. Return
370 ** the final status of load/call with the resulting function (if any)
371 ** in the top of the stack.

373 static int loadline (lua_State *L) {
374  int status;
375  lua_settop(L, 0);
376  if (!pushline(L, 1))
377  return -1; /* no input */
378  if ((status = addreturn(L)) != LUA_OK) /* 'return ...' did not work? */

381  lua_assert(lua_gettop(L) == 1);
382  return status;

402 /*
403 ** Do the REPL: repeatedly read (load) a line, evaluate (call) it, and
404 ** print any results.
405 */
406 static void doREPL (lua_State *L) {
407  int status;
408  const char *oldprogname = progname;
409  progname = NULL; /* no 'progname' on errors in interactive mode */
410  while ((status = loadline(L)) != -1) {
411  if (status == LUA_OK)
412  status = docall(L, 0, LUA_MULTRET);
413  if (status == LUA_OK) l_print(L);
414  else report(L, status);
415  }
416  lua_settop(L, 0); /* clear stack */
417  lua_writeline();
418  progname = oldprogname;
419 }

453 /* bits of various argument indicators in 'args' */
454 #define has_error 1 /* bad option */
455 #define has_i 2 /* -i */
456 #define has_v 4 /* -v */
457 #define has_e 8 /* -e */
458 #define has_E 16 /* -E */

460 /*
461 ** Traverses all arguments from 'argv', returning a mask with those
462 ** needed before running any Lua code (or an error code if it finds
463 ** any invalid argument). 'first' returns the first not-handled argument
464 ** (either the script name or a bad argument in case of error).
465 */
466 static int collectargs (char **argv, int *first) {
467  int args = 0;
468  int i;
469  for (i = 1; argv[i] != NULL; i++) {
470  *first = i;
471  if (argv[i][0] != '-') /* not an option? */
472  return args; /* stop handling options */
473  switch (argv[i][1]) { /* else check option */
474  case '-': /* '--' */
475  if (argv[i][2] != '\0') /* extra characters after '--'? */
476  return has_error; /* invalid option */
477  *first = i + 1;
478  return args;
479  case '\0': /* '-' */
480  return args; /* script "name" is '-' */
481  case 'E':
482  if (argv[i][2] != '\0') /* extra characters after 1st? */
483  return has_error; /* invalid option */
484  args |= has_E;
485  break;
486  case 'i':
487  args |= has_i; /* (-i implies -v) *//* FALLTHROUGH */
488  case 'v':
489  if (argv[i][2] != '\0') /* extra characters after 1st? */
490  return has_error; /* invalid option */
491  args |= has_v;
492  break;
493  case 'e':
494  args |= has_e; /* FALLTHROUGH */
495  case 'l': /* both options need an argument */
496  if (argv[i][2] == '\0') { /* no concatenated argument? */
497  i++; /* try next 'argv' */
498  if (argv[i] == NULL || argv[i][0] == '-')
499  return has_error; /* no next argument or it is another option */
500  }
501  break;
502  default: /* invalid option */
503  return has_error;
504  }
505  }
506  *first = i; /* no script name */
507  return args;
508 }

511 /*
512 ** Processes options 'e' and 'l', which involve running Lua code.
513 ** Returns 0 if some code raises an error.
514 */
515 static int runargs (lua_State *L, char **argv, int n) {
516  int i;
517  for (i = 1; i < n; i++) {
518  int option = argv[i][1];
519  lua_assert(argv[i][0] == '-'); /* already checked */
520  if (option == 'e' || option == 'l') {
521  int status;
522  const char *extra = argv[i] + 2; /* both options need an argument */
523  if (*extra == '\0') extra = argv[++i];
524  lua_assert(extra != NULL);
525  status = (option == 'e')
526  ? dostring(L, extra, "=(command line)")
527  : dolibrary(L, extra);
528  if (status != LUA_OK) return 0;
529  }
530  }
531  return 1;
532 }

551 /*
552 ** Main body of stand-alone interpreter (to be called in protected mode).
553 ** Reads the options and handles them all.
554 */
555 int pmain (lua_State *L) {
556  int argc = (int)lua_tointeger(L, 1);
557  char **argv = (char **)lua_touserdata(L, 2);
558  int script;
559  int args = collectargs(argv, &script);
560  luaL_checkversion(L); /* check that interpreter has correct version */
561  if (argv[0] && argv[0][0]) progname = argv[0];
562  if (args == has_error) { /* bad arg? */
563  print_usage(argv[script]); /* 'script' has index of bad arg. */
564  return 0;
565  }
566  if (args & has_v) /* option '-v'? */
567  print_version();
568  if (args & has_E) { /* option '-E'? */
569  lua_pushboolean(L, 1); /* signal for libraries to ignore env. vars. */
570  lua_setfield(L, LUA_REGISTRYINDEX, "LUA_NOENV");
571  }
572  luaL_openlibs(L); /* open standard libraries */
573  createargtable(L, argv, argc, script); /* create table 'arg' */
574  if (!(args & has_E)) { /* no option '-E'? */
575  if (handle_luainit(L) != LUA_OK) /* run LUA_INIT */
576  return 0; /* error running LUA_INIT */
577  }
578  if (!runargs(L, argv, script)) /* execute arguments -e and -l */
579  return 0; /* something failed */
580  if (script < argc && /* execute main script (if there is one) */
581  handle_script(L, argv + script) != LUA_OK)
582  return 0;
583  if (args & has_i) /* -i option? */
584  doREPL(L); /* do read-eval-print loop */
585  else if (script == argc && !(args & (has_e | has_v))) { /* no arguments? */
586  if (lua_stdin_is_tty()) { /* running in interactive mode? */
587  print_version();
588  doREPL(L); /* do read-eval-print loop */
589  }
590  else dofile(L, NULL); /* executes stdin as a file */
591  }
592  lua_pushboolean(L, 1); /* signal no errors */
593  return 1;
594 }

597 int lua_ymain (int argc, char **argv) {
598  int status, result;
599  lua_State *L = luaL_newstate(); /* create state */
600  if (L == NULL) {
601  l_message(argv[0], "cannot create state: not enough memory");
602  return EXIT_FAILURE;
603  }
604  lua_pushcfunction(L, &pmain); /* to call 'pmain' in protected mode */
605  lua_pushinteger(L, argc); /* 1st argument */
606  lua_pushlightuserdata(L, argv); /* 2nd argument */
607  status = lua_pcall(L, 2, 1, 0); /* do the call */
608  result = lua_toboolean(L, -1); /* get result */
609  report(L, status);
610  lua_close(L);
611  return (result && status == LUA_OK) ? EXIT_SUCCESS : EXIT_FAILURE;
612 }
