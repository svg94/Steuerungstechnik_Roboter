203 BYTE cNetTCP::Socket::process( void )
204 {
205  cNetTCP::Msg &msg = *(cNetTCP::Msg*)tcp.ip.eth.buf;
206 
207  WORD flagsMsg = msg.tcp.getFlags();
208  DWORD ackNumMsg = msg.tcp.acknowledgeNumber;
209  DWORD seqNumMsg = msg.tcp.sequenceNumber;
210 
211  WORD dataLen = msg.ip.packetLength
212  - sizeof(cNetIP_Header)
213  - msg.tcp.getDataOffset();
214 
215  //com.printf("receive:s=%u a=%u f=%x\r\n",seqNumMsg, ackNumMsg, flagsMsg);
216 
217  BYTE *dataPtr = NULL;
218 
219  if( dataLen > 0 )
220  {
221  WORD idx = sizeof( cNetIP_Msg ) + msg.tcp.getDataOffset();
222  if( idx > 0 )
223  {
224  dataPtr = &tcp.ip.eth.buf[ idx ];
225  }
226  }
227 
228  // clear(); /// ??? plen is used later !
229  tcp.ip.eth.plen = 0;
230 
231  if( !(myPort == msg.tcp.destPort) )
232  {
233  return( false );
234  }
235 
236  if( flagsMsg & cNetTCP::FLAG_RST )
237  {
238  //com.printf("\r\nreset\r\n");
239  state = STATE_WAITING;
240  mss = false;
241  return( true );
242  }
243 
244  switch( state )
245  {
246  case STATE_WAITING:
247  if( flagsMsg & cNetTCP::FLAG_SYN )
248  {
249  state = Socket::STATE_PENDING;
250  addr = msg.ip.sourceAddr;
251  remotePort = msg.tcp.sourcePort;
252  addrPhy = msg.eth.sourceAddr;
253  flags = cNetTCP::FLAG_SYN | cNetTCP::FLAG_ACK;
254  mss = true;
255 
256  initialSeqNum += 10000;
257 
258  seqNum = initialSeqNum;
259  ackNum = seqNumMsg + 1;
260 
261  tcp.create( *this, tcp.ip.eth.plen );
262  setTimeout( 3 );
263  return( true) ;
264  }
265  break;
266 
267  case STATE_PENDING:
268  if( flagsMsg & cNetTCP::FLAG_ACK )
269  {
270  state = STATE_CONNECTED;
271  mss = false;
272  }
273  seqNum = seqNum + 1;
274  ackNum = seqNumMsg + 1;
275  break;
276 
277  case STATE_OPEN_WAIT_SERVER:
278  if( flagsMsg & cNetTCP::FLAG_ACK )
279  {
280  state = STATE_CONNECTED;
281  mss = false;
282  flags = cNetTCP::FLAG_ACK;
283  seqNum = seqNum + 1;
284  ackNum = seqNumMsg + 1;
285 
286  tcp.create( *this, tcp.ip.eth.plen );
287  setTimeout( 0 );
288  return( true );
289  }
290  break;
291 
292  case STATE_CONNECTED:
293  if( flagsMsg & cNetTCP::FLAG_FIN )
294  {
295  state = Socket::STATE_FINISHING;
296  flags = cNetTCP::FLAG_FIN | cNetTCP::FLAG_ACK;
297  seqNum = ackNumMsg;
298  ackNum = seqNumMsg + dataLen;
299  tcp.ip.eth.plen = 0; // ???
300 
301  tcp.create( *this, tcp.ip.eth.plen );
302  setTimeout( 3 );
303  return( true );
304  }
305  else
306  {
307  state = Socket::STATE_CONNECTED_WAIT_ACK;
308 
309  flags = cNetTCP::FLAG_ACK;
310  }
311  seqNum = ackNumMsg;
312  ackNum = seqNumMsg + dataLen;
313  break;
314 
315  case STATE_CONNECTED_WAIT_ACK:
316  if( ( flagsMsg & cNetTCP::FLAG_ACK )
317  && ackNumMsg == ackNumExpected )
318  {
319  state = STATE_CONNECTED;
320  flags = cNetTCP::FLAG_NONE;
321  }
322 
323  if( dataLen > 0 )
324  {
325  flags = cNetTCP::FLAG_ACK;
326  }
327  seqNum = ackNumMsg;
328  ackNum = seqNumMsg + dataLen;
329  break;
330 
331  case STATE_FINISHING:
332  if( flagsMsg & cNetTCP::FLAG_FIN )
333  {
334  if( flagsMsg & cNetTCP::FLAG_ACK )
335  {
336  state = Socket::STATE_WAITING;
337  setTimeout(0 );
338  flags = cNetTCP::FLAG_ACK;
339  seqNum = ackNumMsg;
340  ackNum = seqNumMsg + dataLen + 1;
341 
342  tcp.create( *this, tcp.ip.eth.plen );
343  return( true );
344  }
345  else
346  {//com.printf("\r\nreset\r\n");
347  // state = STATE_WAITING;
348 
349  setTimeout( 3 );
350  flags = cNetTCP::FLAG_ACK;
351  seqNum = ackNumMsg + 1;
352  ackNum = seqNumMsg + dataLen + 1;
353 
354  tcp.create( *this, tcp.ip.eth.plen );
355  return( true );
356  }
357  }
358  seqNum = ackNumMsg;
359  ackNum = seqNumMsg + dataLen;
360  break;
361  } // switch
362 
363  if( dataLen > 0 )
364  {
365  seqNum = ackNumMsg;
366  ackNum = seqNumMsg + dataLen;
367 
368  tcp.buf = dataPtr;
369  tcp.inPos = 0;
370  tcp.inBufLen = dataLen;
371 
372  app.OnReceive();
373  }
374 
375  return( true );
376 }
