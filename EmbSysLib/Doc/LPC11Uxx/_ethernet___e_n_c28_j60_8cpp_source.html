431 //*****************************************************************************
432 void cHwEthernet_Enc28j60::Init(const cNetAddr<6> &addrPhyIn)
433 {
434  // perform system reset
435  WriteOp(ENC28J60_SOFT_RESET, 0, ENC28J60_SOFT_RESET);
436  cSystem::delayMilliSec(50);
437  // check CLKRDY bit to see if reset is complete
438  // The CLKRDY does not work. See Rev. B4 Silicon Errata point. Just wait.
439  //while(!(enc28j60Read(ESTAT) & ESTAT_CLKRDY));
440  // do bank 0 stuff
441  // initialize receive buffer
442  // 16-bit transfers, must write low byte first
443  // set receive buffer start address
444  NextPacketPtr = RXSTART_INIT;
445  // Rx start
446  WriteWord(ERXST, RXSTART_INIT );
447  // set receive pointer address
448  WriteWord(ERXRDPT, RXSTART_INIT );
449  // RX end
450  WriteWord(ERXND, RXSTOP_INIT );
451  // TX start
452  WriteWord(ETXST, TXSTART_INIT );
453  // TX end
454  WriteWord(ETXND, TXSTOP_INIT );
455  // do bank 1 stuff, packet filter:
456  // For broadcast packets we allow only ARP packtets
457  // All other packets should be unicast only for our mac (MAADR)
458  //
459  // The pattern to match on is therefore
460  // Type ETH.DST
461  // ARP BROADCAST
462  // 06 08 -- ff ff ff ff ff ff -> ip checksum for theses bytes=f7f9
463  // in binary these poitions are:11 0000 0011 1111
464  // This is hex 303F->EPMM0=0x3f,EPMM1=0x30
465  WriteByte(ERXFCON, 0 /*ERXFCON_UCEN*/ /*| ERXFCON_CRCEN */ /*| ERXFCON_PMEN*/ );
466  WriteByte(EPMM0, 0x3f);
467  WriteByte(EPMM1, 0x30);
468  WriteByte(EPMCSL, 0xf9);
469  WriteByte(EPMCSH, 0xf7);
470  //
471  //
472  // do bank 2 stuff
473  // enable MAC receive
474  WriteByte(MACON1, MACON1_MARXEN | MACON1_TXPAUS | MACON1_RXPAUS);
475  // bring MAC out of reset
476  WriteByte(MACON2, 0x00);
477  // enable automatic padding to 60bytes and CRC operations
478  WriteOp(ENC28J60_BIT_FIELD_SET, MACON3, MACON3_PADCFG0 | MACON3_TXCRCEN | MACON3_FRMLNEN);
479  // set inter-frame gap (non-back-to-back)
480  WriteByte(MAIPGL, 0x12);
481  WriteByte(MAIPGH, 0x0C);
482  // set inter-frame gap (back-to-back)
483  WriteByte(MABBIPG, 0x12);
484  // Set the maximum packet size which the controller will accept
485  // Do not send packets longer than MAX_FRAMELEN:
486  WriteWord(MAMXFL, MAX_FRAMELEN );
487  // do bank 3 stuff
488  // write MAC address
489  // NOTE: MAC address in ENC28J60 is byte-backward
490  WriteByte(MAADR5, addrPhyIn.addr[0]);
491  WriteByte(MAADR4, addrPhyIn.addr[1]);
492  WriteByte(MAADR3, addrPhyIn.addr[2]);
493  WriteByte(MAADR2, addrPhyIn.addr[3]);
494  WriteByte(MAADR1, addrPhyIn.addr[4]);
495  WriteByte(MAADR0, addrPhyIn.addr[5]);
496  // no loopback of transmitted frames
497  PhyWrite(PHCON2, PHCON2_HDLDIS);
498  // switch to bank 0
499  SetBank(ECON1);
500  // enable interrutps
501  WriteOp(ENC28J60_BIT_FIELD_SET, EIE, EIE_INTIE | EIE_PKTIE);
502  // enable packet reception
503  WriteOp(ENC28J60_BIT_FIELD_SET, ECON1, ECON1_RXEN);
504 }

513 void cHwEthernet_Enc28j60::PacketSend(uint8_t* packet, uint16_t len)
514 {
515  // Set the write pointer to start of transmit buffer area
516  WriteWord(EWRPT, TXSTART_INIT );
517  // Set the TXND pointer to correspond to the packet size given
518  WriteWord(ETXND, (TXSTART_INIT + len) );
519  // write per-packet control byte (0x00 means use macon3 settings)
520  WriteOp(ENC28J60_WRITE_BUF_MEM, 0, 0x00);
521  // copy the packet into the transmit buffer
522  WriteBuffer(len, packet);
523  // send the contents of the transmit buffer onto the network
524  WriteOp(ENC28J60_BIT_FIELD_SET, ECON1, ECON1_TXRTS);
525  // Reset the transmit logic problem. See Rev. B4 Silicon Errata point 12.
526  if((Read(EIR) & EIR_TXERIF))
527  {
529  Init(addrPhy);
530  cSystem::delayMilliSec(10);
531  InitPhy();
532 // WriteOp(ENC28J60_BIT_FIELD_CLR, ECON1, ECON1_TXRTS);
533  }
534 }

544 uint16_t maxlen = BUFFER_SIZE;
545  uint16_t rxstat;
546  uint16_t len;
547  // check if a packet has been received and buffered
548  //if( !(enc28j60Read(EIR) & EIR_PKTIF) ){
549  // The above does not work. See Rev. B4 Silicon Errata point 6.
550  if( Read(EPKTCNT) ==0 )
551  {
552  return(0);
553  }
554 
555  // Set the read pointer to the start of the received packet
556  WriteWord(ERDPT, NextPacketPtr);
557 
558  // read the next packet pointer
559  NextPacketPtr = ReadOp(ENC28J60_READ_BUF_MEM, 0);
560  NextPacketPtr |= ReadOp(ENC28J60_READ_BUF_MEM, 0) << 8;
561 
562  // read the packet length (see datasheet page 43)
563  len = ReadOp(ENC28J60_READ_BUF_MEM, 0);
564  len |= ReadOp(ENC28J60_READ_BUF_MEM, 0) << 8;
565  len -= 4; //remove the CRC count
566 
567  // read the receive status (see datasheet page 43)
568  rxstat = ReadOp(ENC28J60_READ_BUF_MEM, 0);
569  rxstat |= ReadOp(ENC28J60_READ_BUF_MEM, 0) << 8;
570 
571  // limit retrieve length
572  if (len>maxlen-1)
573  {
574  len=maxlen-1;
575  }
576  // check CRC and symbol errors (see datasheet page 44, table 7-3):
577  // The ERXFCON.CRCEN is set by default. Normally we should not
578  // need to check this.
579  if ((rxstat & 0x80)==0)
580  {
581  // invalid
582  len=0;
583  }
584  else
585  {
586  // copy the packet from the receive buffer
587  ReadBuffer(len, buf);
588  }
589 
590  // Move the RX read pointer to the start of the next received packet
591  // This frees the memory we just read out
592  WriteWord(ERXRDPT, NextPacketPtr);
593 
594  // decrement the packet counter indicate we are done with this packet
595  WriteOp(ENC28J60_BIT_FIELD_SET, ECON2, ECON2_PKTDEC);
596 
597  plen = len;
598  return( (plen==0)?false:true );
599 }
