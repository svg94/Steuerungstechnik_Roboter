369 void SystemCoreClockUpdate (void) /* Get Core Clock Frequency */
370 {
371  uint32_t oscClk = 0;
372 
373  /* Determine clock frequency according to clock register values */
374  switch ((LPC_SYSCON->WDTOSCCTRL >> 5) & 0x0F) {
375  case 0: oscClk = 0; break;
376  case 1: oscClk = 500000; break;
377  case 2: oscClk = 800000; break;
378  case 3: oscClk = 1100000; break;
379  case 4: oscClk = 1400000; break;
380  case 5: oscClk = 1600000; break;
381  case 6: oscClk = 1800000; break;
382  case 7: oscClk = 2000000; break;
383  case 8: oscClk = 2200000; break;
384  case 9: oscClk = 2400000; break;
385  case 10: oscClk = 2600000; break;
386  case 11: oscClk = 2700000; break;
387  case 12: oscClk = 2900000; break;
388  case 13: oscClk = 3100000; break;
389  case 14: oscClk = 3200000; break;
390  case 15: oscClk = 3400000; break;
391  }
392  oscClk /= ((LPC_SYSCON->WDTOSCCTRL & 0x1F) << 1) + 2;
393 
394  switch (LPC_SYSCON->MAINCLKSEL & 0x03) {
395  case 0: /* Internal RC oscillator */
396  SystemCoreClock = __IRC_OSC_CLK;
397  break;
398  case 1: /* Input Clock to System PLL */
399  switch (LPC_SYSCON->SYSPLLCLKSEL & 0x03) {
400  case 0: /* Internal RC oscillator */
401  SystemCoreClock = __IRC_OSC_CLK;
402  break;
403  case 1: /* System oscillator */
404  SystemCoreClock = __SYS_OSC_CLK;
405  break;
406  case 2: /* Reserved */
407  case 3: /* Reserved */
408  SystemCoreClock = 0;
409  break;
410  }
411  break;
412  case 2: /* WDT Oscillator */
413  SystemCoreClock = oscClk;
414  break;
415  case 3: /* System PLL Clock Out */
416  switch (LPC_SYSCON->SYSPLLCLKSEL & 0x03) {
417  case 0: /* Internal RC oscillator */
418  if (LPC_SYSCON->SYSPLLCTRL & 0x180) {
419  SystemCoreClock = __IRC_OSC_CLK;
420  } else {
421  SystemCoreClock = __IRC_OSC_CLK * ((LPC_SYSCON->SYSPLLCTRL & 0x01F) + 1);
422  }
423  break;
424  case 1: /* System oscillator */
425  if (LPC_SYSCON->SYSPLLCTRL & 0x180) {
426  SystemCoreClock = __SYS_OSC_CLK;
427  } else {
428  SystemCoreClock = __SYS_OSC_CLK * ((LPC_SYSCON->SYSPLLCTRL & 0x01F) + 1);
429  }
430  break;
431  case 2: /* Reserved */
432  case 3: /* Reserved */
433  SystemCoreClock = 0;
434  break;
435  }
436  break;
437  }
438 
439  SystemCoreClock /= LPC_SYSCON->SYSAHBCLKDIV;
440
