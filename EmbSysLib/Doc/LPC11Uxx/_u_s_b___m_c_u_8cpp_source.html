492 WORD cHwUSB_0::readEP( BYTE epAddr,
493  BYTE *dataPtr,
494  WORD len )
495 {
496  uint32_t cnt=0;
497  uint32_t epIdx = epAddr2Index(epAddr);
498  DWORD *addr = (DWORD*)(USB_EPLIST_ADDR + (epIdx<<3));
499  DWORD *ptr;
500 
501  if( !(epAddr & 0x80) )
502  {
503  if( (epIdx == 0) && isSetup ) // setup ?
504  {
505  addr++;
506  cnt = USB_SETUP_PACKET_SIZE;
507  ptr = (DWORD *)EPList[1].buf_ptr;
508  }
509  else
510  {
511  cnt = (*addr >> 16) & PKT_LNGTH_MASK;
512  /* The NBytes field decrements by H/W with the packet byte each time. */
513  cnt = EPList[epIdx*2].buf_length - cnt;
514  ptr = (DWORD *)EPList[epIdx*2].buf_ptr;
515  }
516 
517  cnt = (cnt<len)?cnt:len;
518 
519  memcpy (dataPtr, ptr, cnt);
520 
521  /* Clear buffer after EP read, reset EP length and buffer pointer field */
522  if( (epIdx == 0) && isSetup ) // setup ?
523  {
524  *addr &= ~0x3FFFFFF;
525  *addr |= (uint16_t)((EPList[1].buf_ptr)>>6);
526  }
527  else
528  {
529  *addr &= ~0x3FFFFFF;
530  *addr |= ((EPList[epIdx*2].buf_length<<16)
531  | (uint16_t)((EPList[epIdx*2].buf_ptr)>>6) | BUF_ACTIVE);
532  }
533  return (cnt);
534  }
535  return( 0 );
536 }
