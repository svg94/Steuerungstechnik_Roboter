73 /*
74 ** Compute an initial seed as random as possible. Rely on Address Space
75 ** Layout Randomization (if present) to increase randomness..
76 */
77 #define addbuff(b,p,e) \
78  { size_t t = cast(size_t, e); \
79  memcpy(b + p, &t, sizeof(t)); p += sizeof(t); }

81 static unsigned int makeseed (lua_State *L) {
82  char buff[4 * sizeof(size_t)];
83  unsigned int h = luai_makeseed();
84  int p = 0;
85  addbuff(buff, p, L); /* heap variable */
86  addbuff(buff, p, &h); /* local variable */
87  addbuff(buff, p, luaO_nilobject); /* global variable */
88  addbuff(buff, p, &lua_newstate); /* public function */
89  // addbuff(buff, p, &lua_setParent); /* public function */
90  lua_assert(p == sizeof(buff));
91  return luaS_hash(buff, p, h);
92 }

95 /*
96 ** set GCdebt to a new value keeping the value (totalbytes + GCdebt)
97 ** invariant (and avoiding underflows in 'totalbytes')
98 */
99 void luaE_setdebt (global_State *g, l_mem debt) {
100  l_mem tb = gettotalbytes(g);
101  lua_assert(tb > 0);
102  if (debt < tb - MAX_LMEM)
103  debt = tb - MAX_LMEM; /* will make 'totalbytes == MAX_LMEM' */
104  g->totalbytes = tb - debt;
105  g->GCdebt = debt;
106 }

123 void luaE_freeCI (lua_State *L) {

125  CallInfo *next = ci->next;

138 void luaE_shrinkCI (lua_State *L) {
139  CallInfo *ci = L->ci;
140  CallInfo *next2; /* next's next */
141  /* while there are two nexts */
142  while (ci->next != NULL && (next2 = ci->next->next) != NULL) {
143  luaM_free(L, ci->next); /* free next */
144  L->nci--;
145  ci->next = next2; /* remove 'next' from the list */
146  next2->previous = ci;
147  ci = next2; /* keep next's next */
148  }
149 }

152 static void stack_init (lua_State *L1, lua_State *L) {
153  int i; CallInfo *ci;
154  /* initialize stack array */
155  L1->stack = luaM_newvector(L, BASIC_STACK_SIZE, TValue);
156  L1->stacksize = BASIC_STACK_SIZE;
157  for (i = 0; i < BASIC_STACK_SIZE; i++)
158  setnilvalue(L1->stack + i); /* erase new stack */
159  L1->top = L1->stack;
160  L1->stack_last = L1->stack + L1->stacksize - EXTRA_STACK;
161  /* initialize first ci */
162  ci = &L1->base_ci;
163  ci->next = ci->previous = NULL;
164  ci->callstatus = 0;
165  ci->func = L1->top;
166  setnilvalue(L1->top++); /* 'function' entry for this 'ci' */
167  ci->top = L1->top + LUA_MINSTACK;
168  L1->ci = ci;
169 }

172 static void freestack (lua_State *L) {
173  if (L->stack == NULL)
174  return; /* stack not completely built yet */
175  L->ci = &L->base_ci; /* free the entire 'ci' list */
176  luaE_freeCI(L);
177  lua_assert(L->nci == 0);
178  luaM_freearray(L, L->stack, L->stacksize); /* free stack array */
179 }

182 /*
183 ** Create registry table and its predefined values
184 */
185 static void init_registry (lua_State *L, global_State *g) {
186  TValue temp;
187  /* create registry */
188  Table *registry = luaH_new(L);
189  sethvalue(L, &g->l_registry, registry);
190  luaH_resize(L, registry, LUA_RIDX_LAST, 0);
191  /* registry[LUA_RIDX_MAINTHREAD] = L */
192  setthvalue(L, &temp, L); /* temp = L */
193  luaH_setint(L, registry, LUA_RIDX_MAINTHREAD, &temp);
194  /* registry[LUA_RIDX_GLOBALS] = table of globals */
195  sethvalue(L, &temp, luaH_new(L)); /* temp = new table (global table) */
196  luaH_setint(L, registry, LUA_RIDX_GLOBALS, &temp);
197 }

243 static void close_state (lua_State *L) {
244  global_State *g = G(L);
245  luaF_close(L, L->stack); /* close all upvalues for this thread */
246  luaC_freeallobjects(L); /* collect all objects */
247  if (g->version) /* closing a fully built state? */
248  luai_userstateclose(L);
249  luaM_freearray(L, G(L)->strt.hash, G(L)->strt.size);
250  freestack(L);
251  lua_assert(gettotalbytes(g) == sizeof(LG));
252  (*g->frealloc)(g->ud, fromstate(L), sizeof(LG), 0); /* free main block */
253 }

256 LUA_API lua_State *lua_newthread (lua_State *L) {
257  global_State *g = G(L);
258  lua_State *L1;
259  lua_lock(L);
260  luaC_checkGC(L);
261  /* create new thread */
262  L1 = &cast(LX *, luaM_newobject(L, LUA_TTHREAD, sizeof(LX)))->l;
263  L1->marked = luaC_white(g);
264  L1->tt = LUA_TTHREAD;
265  /* link it on list 'allgc' */
266  L1->next = g->allgc;
267  g->allgc = obj2gco(L1);
268  /* anchor it on L stack */
269  setthvalue(L, L->top, L1);
270  api_incr_top(L);
271  preinit_thread(L1, g);
272  L1->hookmask = L->hookmask;
273  L1->basehookcount = L->basehookcount;
274  L1->hook = L->hook;
275  resethookcount(L1);
276  /* initialize L1 extra space */
277  memcpy(lua_getextraspace(L1), lua_getextraspace(g->mainthread),
278  LUA_EXTRASPACE);
279  luai_userstatethread(L, L1);
280  stack_init(L1, L); /* init stack */
281  lua_unlock(L);
282  return L1;
283 }

296 LUA_API lua_State *lua_newstate (lua_Alloc f, void *ud) {
297  int i;
298  lua_State *L;
299  global_State *g;
300  LG *l = cast(LG *, (*f)(ud, NULL, LUA_TTHREAD, sizeof(LG)));
301  if (l == NULL) return NULL;
302  L = &l->l.l;
303  g = &l->g;
304  L->next = NULL;
305  L->tt = LUA_TTHREAD;
306  g->currentwhite = bitmask(WHITE0BIT);
307  L->marked = luaC_white(g);
308  preinit_thread(L, g);
309  g->frealloc = f;
310  g->ud = ud;
311  g->mainthread = L;
312  g->seed = makeseed(L);
313  g->gcrunning = 0; /* no GC while building state */
314  g->GCestimate = 0;
315  g->strt.size = g->strt.nuse = 0;
316  g->strt.hash = NULL;
317  setnilvalue(&g->l_registry);
318  g->panic = NULL;
319  g->version = NULL;
320  g->gcstate = GCSpause;
321  g->gckind = KGC_NORMAL;
322  g->allgc = g->finobj = g->tobefnz = g->fixedgc = NULL;
323  g->sweepgc = NULL;
324  g->gray = g->grayagain = NULL;
325  g->weak = g->ephemeron = g->allweak = NULL;
326  g->twups = NULL;
327  g->totalbytes = sizeof(LG);
328  g->GCdebt = 0;
329  g->gcfinnum = 0;
330  g->gcpause = LUAI_GCPAUSE;
331  g->gcstepmul = LUAI_GCMUL;
332  for (i=0; i < LUA_NUMTAGS; i++) g->mt[i] = NULL;
333  if (luaD_rawrunprotected(L, f_luaopen, NULL) != LUA_OK) {
334  /* memory allocation error: free partial state */
335  close_state(L);
336  L = NULL;
337  }
338  return L;
339 }
