{

while( 1 )
{

switch( encoder.get() )
{

case cDevControlEncoder::RIGHT: cnt += 1; break;
case cDevControlEncoder::CTRL_DWN: cnt = 0; btn = 1; break;
case cDevControlEncoder::CTRL_UP: btn = 0; break;

}
