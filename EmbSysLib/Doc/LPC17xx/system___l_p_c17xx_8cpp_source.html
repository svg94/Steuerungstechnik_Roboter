399 void SystemInit (void)
400 {
401 #if (CLOCK_SETUP) /* Clock Setup */
402  LPC_SC->SCS = SCS_Val;
403  if (SCS_Val & (1 << 5)) { /* If Main Oscillator is enabled */
404  while ((LPC_SC->SCS & (1<<6)) == 0);/* Wait for Oscillator to be ready */
405  }
406 
407  LPC_SC->CCLKCFG = CCLKCFG_Val; /* Setup Clock Divider */
408 
409  LPC_SC->PCLKSEL0 = PCLKSEL0_Val; /* Peripheral Clock Selection */
410  LPC_SC->PCLKSEL1 = PCLKSEL1_Val;
411 
412  LPC_SC->CLKSRCSEL = CLKSRCSEL_Val; /* Select Clock Source for PLL0 */
413 
414 #if (PLL0_SETUP)
415  LPC_SC->PLL0CFG = PLL0CFG_Val;
416  LPC_SC->PLL0CON = 0x01; /* PLL0 Enable */
417  LPC_SC->PLL0FEED = 0xAA;
418  LPC_SC->PLL0FEED = 0x55;
419  while (!(LPC_SC->PLL0STAT & (1<<26)));/* Wait for PLOCK0 */
420 
421  LPC_SC->PLL0CON = 0x03; /* PLL0 Enable & Connect */
422  LPC_SC->PLL0FEED = 0xAA;
423  LPC_SC->PLL0FEED = 0x55;
424 #endif
425 
426 #if (PLL1_SETUP)
427  LPC_SC->PLL1CFG = PLL1CFG_Val;
428  LPC_SC->PLL1CON = 0x01; /* PLL1 Enable */
429  LPC_SC->PLL1FEED = 0xAA;
430  LPC_SC->PLL1FEED = 0x55;
431  while (!(LPC_SC->PLL1STAT & (1<<10)));/* Wait for PLOCK1 */
432 
433  LPC_SC->PLL1CON = 0x03; /* PLL1 Enable & Connect */
434  LPC_SC->PLL1FEED = 0xAA;
435  LPC_SC->PLL1FEED = 0x55;
436 #else
437  LPC_SC->USBCLKCFG = USBCLKCFG_Val; /* Setup USB Clock Divider */
438 #endif
439 
440  LPC_SC->PCONP = PCONP_Val; /* Power Control for Peripherals */
441 
442  LPC_SC->CLKOUTCFG = CLKOUTCFG_Val; /* Clock Output Configuration */
443 #endif
444 
445  /* Determine clock frequency according to clock register values */
446  if (((LPC_SC->PLL0STAT >> 24)&3)==3) {/* If PLL0 enabled and connected */
447  switch (LPC_SC->CLKSRCSEL & 0x03) {
448  case 0: /* Internal RC oscillator => PLL0 */
449  case 3: /* Reserved, default to Internal RC */
450  SystemFrequency = (IRC_OSC *
451  ((2 * ((LPC_SC->PLL0STAT & 0x7FFF) + 1))) /
452  (((LPC_SC->PLL0STAT >> 16) & 0xFF) + 1) /
453  ((LPC_SC->CCLKCFG & 0xFF)+ 1));
454  break;
455  case 1: /* Main oscillator => PLL0 */
456  SystemFrequency = (OSC_CLK *
457  ((2 * ((LPC_SC->PLL0STAT & 0x7FFF) + 1))) /
458  (((LPC_SC->PLL0STAT >> 16) & 0xFF) + 1) /
459  ((LPC_SC->CCLKCFG & 0xFF)+ 1));
460  break;
461  case 2: /* RTC oscillator => PLL0 */
462  SystemFrequency = (RTC_CLK *
463  ((2 * ((LPC_SC->PLL0STAT & 0x7FFF) + 1))) /
464  (((LPC_SC->PLL0STAT >> 16) & 0xFF) + 1) /
465  ((LPC_SC->CCLKCFG & 0xFF)+ 1));
466  break;
467  }
468  } else {
469  switch (LPC_SC->CLKSRCSEL & 0x03) {
470  case 0: /* Internal RC oscillator => PLL0 */
471  case 3: /* Reserved, default to Internal RC */
472  SystemFrequency = IRC_OSC / ((LPC_SC->CCLKCFG & 0xFF)+ 1);
473  break;
474  case 1: /* Main oscillator => PLL0 */
475  SystemFrequency = OSC_CLK / ((LPC_SC->CCLKCFG & 0xFF)+ 1);
476  break;
477  case 2: /* RTC oscillator => PLL0 */
478  SystemFrequency = RTC_CLK / ((LPC_SC->CCLKCFG & 0xFF)+ 1);
479  break;
480  }
481  }
482 
483 #if (FLASH_SETUP == 1) /* Flash Accelerator Setup */
484  LPC_SC->FLASHCFG = (LPC_SC->FLASHCFG & ~0x0000F000) | FLASHCFG_Val;
485 #endif
486 }
